// Simple 32-bit register
module reg32 (clk, reset, d_in, d_out);
    input               clk, reset;
    input   [31:0]	d_in;
    output  [31:0] 	d_out;
    reg     [31:0]	d_out;
   
    always @(posedge clk)
    begin
        if (reset) d_out <= 0;
        else d_out <= d_in;
    end

endmodule
